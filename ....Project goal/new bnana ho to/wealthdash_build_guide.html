🗄️ Supabase Database Schema

Supabase Dashboard mein jaao → SQL Editor → yeh SQL paste karke run karo. Ek baar mein sab tables ban jaayengi.

⚠️
Supabase pe account banao (supabase.com), new project create karo, phir SQL Editor mein neeche diya hua SQL run karo.
001 — Users & Core Tables
-- Users table (Supabase auth ke saath link)
CREATE TABLE profiles (
  id          UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
  full_name   TEXT,
  phone       TEXT,
  pan_masked  TEXT,              -- Last 4 chars only
  tax_slab    NUMERIC(5,2) DEFAULT 30,  -- 5/10/15/20/25/30
  tax_regime  TEXT DEFAULT 'new',   -- old/new
  risk_profile TEXT DEFAULT 'moderate',
  monthly_expense NUMERIC(12,2),
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- App settings (key-value store per user)
CREATE TABLE app_settings (
  id          BIGSERIAL PRIMARY KEY,
  user_id     UUID REFERENCES profiles(id) ON DELETE CASCADE,
  key         TEXT NOT NULL,
  value       TEXT,
  UNIQUE(user_id, key)
);
002 — Mutual Funds Tables
-- MF Master (fund info)
CREATE TABLE mf_master (
  scheme_code  INTEGER PRIMARY KEY,   -- AMFI scheme code
  scheme_name  TEXT NOT NULL,
  fund_house   TEXT,
  category     TEXT,               -- Equity/Debt/Hybrid
  sub_category TEXT,
  nav_latest   NUMERIC(10,4),
  nav_date     DATE,
  expense_ratio NUMERIC(5,4),
  updated_at   TIMESTAMPTZ DEFAULT NOW()
);

-- User MF Transactions
CREATE TABLE mf_transactions (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  scheme_code  INTEGER REFERENCES mf_master(scheme_code),
  txn_type     TEXT NOT NULL,         -- BUY/SELL/SIP/SWP/SWITCH_IN/SWITCH_OUT/DIV
  txn_date     DATE NOT NULL,
  nav          NUMERIC(10,4) NOT NULL,
  units        NUMERIC(12,4) NOT NULL,
  amount       NUMERIC(15,2) NOT NULL,
  folio_number TEXT,
  notes        TEXT,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- SIP Rules
CREATE TABLE sip_rules (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  scheme_code  INTEGER REFERENCES mf_master(scheme_code),
  amount       NUMERIC(12,2),
  frequency    TEXT DEFAULT 'monthly',
  sip_date     INTEGER,               -- Day of month: 1-28
  start_date   DATE,
  end_date     DATE,
  is_active    BOOLEAN DEFAULT TRUE,
  goal_id      BIGINT               -- Link to goals table
);

-- MF NAV History (for charts)
CREATE TABLE nav_history (
  scheme_code  INTEGER,
  nav_date     DATE,
  nav          NUMERIC(10,4),
  PRIMARY KEY(scheme_code, nav_date)
);

-- Indexes for speed
CREATE INDEX idx_mf_txn_user ON mf_transactions(user_id);
CREATE INDEX idx_mf_txn_scheme ON mf_transactions(scheme_code);
CREATE INDEX idx_nav_history ON nav_history(scheme_code, nav_date DESC);
003 — Stocks, FD, Crypto Tables
-- Stocks
CREATE TABLE stocks_transactions (
  id          BIGSERIAL PRIMARY KEY,
  user_id     UUID REFERENCES profiles(id) ON DELETE CASCADE,
  symbol      TEXT NOT NULL,          -- RELIANCE, TCS, etc.
  exchange    TEXT DEFAULT 'NSE',
  txn_type    TEXT NOT NULL,          -- BUY/SELL/DIV/BONUS/SPLIT
  txn_date    DATE NOT NULL,
  quantity    NUMERIC(12,4),
  price       NUMERIC(12,4),
  amount      NUMERIC(15,2),
  broker      TEXT,
  notes       TEXT
);

-- Fixed Deposits
CREATE TABLE fd_accounts (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  bank_name    TEXT NOT NULL,
  fd_number    TEXT,
  principal    NUMERIC(15,2) NOT NULL,
  rate         NUMERIC(5,4) NOT NULL,    -- Annual rate e.g. 7.50
  start_date   DATE NOT NULL,
  maturity_date DATE NOT NULL,
  interest_type TEXT DEFAULT 'cumulative', -- cumulative/monthly/quarterly
  is_active    BOOLEAN DEFAULT TRUE,
  notes        TEXT
);

-- Crypto Holdings
CREATE TABLE crypto_transactions (
  id          BIGSERIAL PRIMARY KEY,
  user_id     UUID REFERENCES profiles(id) ON DELETE CASCADE,
  coin_id     TEXT NOT NULL,           -- bitcoin, ethereum (CoinGecko ID)
  symbol      TEXT NOT NULL,           -- BTC, ETH
  txn_type    TEXT NOT NULL,           -- BUY/SELL/TRANSFER
  txn_date    DATE NOT NULL,
  quantity    NUMERIC(18,8),
  price_inr   NUMERIC(15,2),
  amount_inr  NUMERIC(15,2),
  exchange    TEXT                      -- WazirX, Binance, etc.
);
004 — Goals, NPS, Insurance, Loans
-- Goals
CREATE TABLE goals (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  goal_name    TEXT NOT NULL,          -- Retirement, Child Education, etc.
  target_amount NUMERIC(15,2),
  target_date  DATE,
  current_value NUMERIC(15,2) DEFAULT 0,
  category     TEXT,                   -- retirement/education/house/emergency/other
  icon         TEXT,
  is_active    BOOLEAN DEFAULT TRUE
);

-- NPS Accounts
CREATE TABLE nps_accounts (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  pran         TEXT,
  pfm_name     TEXT,                   -- SBI, HDFC, Kotak, etc.
  tier         TEXT DEFAULT 'tier1',
  equity_pct   NUMERIC(5,2),
  corporate_pct NUMERIC(5,2),
  govt_pct     NUMERIC(5,2),
  current_value NUMERIC(15,2)
);

-- Insurance Policies
CREATE TABLE insurance_policies (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  policy_type  TEXT NOT NULL,          -- term/health/ulip/vehicle/other
  insurer      TEXT,
  policy_number TEXT,
  sum_assured  NUMERIC(15,2),
  premium      NUMERIC(12,2),
  premium_freq TEXT,                  -- annual/monthly/quarterly
  start_date   DATE,
  renewal_date DATE,
  is_active    BOOLEAN DEFAULT TRUE
);

-- Loans
CREATE TABLE loans (
  id           BIGSERIAL PRIMARY KEY,
  user_id      UUID REFERENCES profiles(id) ON DELETE CASCADE,
  loan_type    TEXT NOT NULL,          -- home/personal/education/vehicle/other
  lender       TEXT,
  principal    NUMERIC(15,2),
  outstanding  NUMERIC(15,2),
  interest_rate NUMERIC(5,4),
  emi          NUMERIC(12,2),
  start_date   DATE,
  end_date     DATE,
  is_active    BOOLEAN DEFAULT TRUE
);

-- Row Level Security (RLS) — IMPORTANT: Har user sirf apna data dekhe
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE mf_transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE stocks_transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE fd_accounts ENABLE ROW LEVEL SECURITY;
ALTER TABLE crypto_transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE goals ENABLE ROW LEVEL SECURITY;
ALTER TABLE insurance_policies ENABLE ROW LEVEL SECURITY;
ALTER TABLE loans ENABLE ROW LEVEL SECURITY;

-- RLS Policies (users apna hi data dekh/edit kar sakte hain)
CREATE POLICY "users_own_data" ON mf_transactions
  FOR ALL USING (auth.uid() = user_id);

Here is the JavaScript port of our existing PHP XIRR: function xirr(cashflows, dates) { // cashflows: negative for invest, positive for current value // Newton-Raphson iteration let rate = 0.1; for (let i = 0; i < 100; i++) { let f = 0, df = 0; for (let j = 0; j < cashflows.length; j++) { const t = (dates[j] - dates[0]) / 365; f += cashflows[j] / Math.pow(1 + rate, t); df -= t * cashflows[j] / Math.pow(1 + rate, t + 1); } const newRate = rate - f / df; if (Math.abs(newRate - rate) < 1e-6) return newRate; rate = newRate; } return rate; } Save this in lib/calculations/xirr.ts. API ROUTES needed: - app/api/mf/nav/route.ts → fetch latest NAV from AMFI: https://www.amfiindia.com/spages/NAVAll.txt - app/api/mf/search/route.ts → search funds from mf_master table Data from Supabase tables: mf_transactions, mf_master.
